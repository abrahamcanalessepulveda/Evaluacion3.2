from flask import Flask, render_template, request, redirect, url_for

app = Flask(__name__)


@app.route('/')
def index():
    """Ruta de la página principal (menú con botones)."""
    # Muestra página principal
    return render_template('index.html')


# =================================================================
# EJERCICIO 1:(Aprobado/Reprobado)
# =================================================================
@app.route('/ejercicio1', methods=['GET', 'POST'])
def ejercicio1():
    """ 
    Maneja el formulario GET y el procesamiento POST para el Ejercicio 1. 
    """
    if request.method == 'POST':
        try:
            # Insertar datos del las 3  notas
            nota1 = float(request.form.get('nota1'))
            nota2 = float(request.form.get('nota2'))
            nota3 = float(request.form.get('nota3'))
            asistencia = float(request.form.get('asistencia'))

            # calcular promedio notas
            promedio = (nota1 + nota2 + nota3) / 3

            # Aprobado/Reprobado)
            # Aprobado si: Promedio >= 40 Y Asistencia >= 75%
            if promedio >= 40 and asistencia >= 75:
                estado = "APROBADO"
            else:
                estado = "REPROBADO"

            # Renderizar la  página con resultados
            return render_template('ejercicio1_form.html',
                                   promedio=f"{promedio:.1f}",
                                   estado=estado)

        except ValueError:
            # entradas no son números ERROR
            return render_template('ejercicio1_form.html', error="Por favor, ingrese valores numéricos válidos.")

    # Mostrar formulario GET inicial
    return render_template('ejercicio1_form.html')


# =================================================================
# 2 NOMBRE CON MÁS CARACTERES
# =================================================================
@app.route('/ejercicio2', methods=['GET', 'POST'])
def ejercicio2():
    """ 
    Maneja el formulario GET y el procesamiento POST para el Ejercicio 2. 
    """
    if request.method == 'POST':
        # Agregar 3 nombres
        nombre1 = request.form.get('nombre1', '').strip()
        nombre2 = request.form.get('nombre2', '').strip()
        nombre3 = request.form.get('nombre3', '').strip()

        nombres = [nombre1, nombre2, nombre3]

        # encontrar nombre más largo
        nombre_mas_largo = ""
        max_caracteres = -1

        # Encontrar  nombre con la mayor cantidad de caracteres
        for nombre in nombres:
            cantidad_caracteres = len(nombre)
            if cantidad_caracteres > max_caracteres:
                max_caracteres = cantidad_caracteres
                nombre_mas_largo = nombre

        # Generar el resultado a mostrar
        if nombre_mas_largo:
            resultado_nombre = f"El nombre con mayor cantidad de caracteres es: {nombre_mas_largo}"
            resultado_caracteres = f"El nombre tiene: {max_caracteres} caracteres"
        else:
            resultado_nombre = "No se ingresaron nombres válidos."
            resultado_caracteres = ""

        # Renderizar página con resultados
        return render_template('ejercicio2_form.html',
                               resultado_nombre=resultado_nombre,
                               resultado_caracteres=resultado_caracteres)

    # Mostrar formulario GET inicial
    return render_template('ejercicio2_form.html')


if __name__ == '__main__':
    # Ejecuta aplicación en modo debug
    app.run(debug=True)
